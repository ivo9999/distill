# Distill v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Distill — a Discord bot + web dashboard that turns Discord discussions into weekly newsletter drafts, with Stripe billing and multi-platform publishing.

**Architecture:** Monorepo with two apps: Go backend (3 binaries: API server, Discord bot, River worker) and Next.js frontend (dashboard + LLM pipeline). They share a Postgres database and communicate via internal HTTP endpoints authenticated with a shared secret. An experiment CLI script reuses the same LLM pipeline module.

**Tech Stack:** Go 1.22+ (chi, discordgo, sqlc, river), Next.js 15 (App Router, NextAuth v5, Vercel AI SDK, shadcn/ui, Tailwind), PostgreSQL 16, Stripe, Docker Compose.

**Reference:** All requirements come from `docs/prd.md`. Section numbers (4.1–4.9) map to task groups below.

```sql
CREATE TABLE users (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    discord_id          TEXT NOT NULL UNIQUE,
    discord_username    TEXT NOT NULL,
    email               TEXT NOT NULL,
    avatar_url          TEXT,
    stripe_customer_id  TEXT UNIQUE,
    subscription_status TEXT NOT NULL DEFAULT 'trialing',
    trial_ends_at       TIMESTAMPTZ NOT NULL DEFAULT (NOW() + INTERVAL '14 days'),
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE TABLE servers (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id             UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    discord_guild_id    TEXT NOT NULL,
    name                TEXT NOT NULL,
    icon_url            TEXT,
    community_type      TEXT,
    schedule_cron       TEXT NOT NULL DEFAULT '0 18 * * 0',
    status              TEXT NOT NULL DEFAULT 'active',
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE (user_id, discord_guild_id)
);

CREATE TABLE monitored_channels (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    server_id           UUID NOT NULL REFERENCES servers(id) ON DELETE CASCADE,
    discord_channel_id  TEXT NOT NULL,
    name                TEXT NOT NULL,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),
    UNIQUE (server_id, discord_channel_id)
);

CREATE TABLE messages (
    id                      UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    discord_message_id      TEXT NOT NULL UNIQUE,
    server_id               UUID NOT NULL REFERENCES servers(id) ON DELETE CASCADE,
    channel_id              UUID NOT NULL REFERENCES monitored_channels(id) ON DELETE CASCADE,
    discord_author_id       TEXT NOT NULL,
    author_display_name     TEXT NOT NULL,
    content                 TEXT NOT NULL,
    reply_to_discord_id     TEXT,
    thread_discord_id       TEXT,
    sent_at                 TIMESTAMPTZ NOT NULL,
    reaction_count          INTEGER NOT NULL DEFAULT 0,
    reply_count             INTEGER NOT NULL DEFAULT 0,
    raw_payload             JSONB NOT NULL,
    created_at              TIMESTAMPTZ NOT NULL DEFAULT NOW()
);
CREATE INDEX idx_messages_server_sent ON messages(server_id, sent_at DESC);
CREATE INDEX idx_messages_channel_sent ON messages(channel_id, sent_at DESC);

CREATE TABLE optouts (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    server_id           UUID NOT NULL REFERENCES servers(id) ON DELETE CASCADE,
    discord_user_id     TEXT NOT NULL,
    created_at          TIMESTAMPTZ NOT NULL DEFAULT NOW(),

CREATE TABLE newsletters (

    server_id           UUID NOT NULL REFERENCES servers(id) ON DELETE CASCADE,

    period_end          TIMESTAMPTZ NOT NULL,
    status              TEXT NOT NULL DEFAULT 'draft',
    draft_markdown      TEXT NOT NULL,
    edited_markdown     TEXT,
    published_at        TIMESTAMPTZ,
    published_url       TEXT,

    pass1_tokens_out    INTEGER NOT NULL DEFAULT 0,
    pass2_tokens_in     INTEGER NOT NULL DEFAULT 0,
    pass2_tokens_out    INTEGER NOT NULL DEFAULT 0,

CREATE INDEX idx_newsletters_server_created ON newsletters(server_id, created_at DESC);
```

```go
// Config holds all environment configuration.

		DiscordClientID:     os.Getenv("DISCORD_CLIENT_ID"),
```

- [ ] **Step 2: Create encryption helpers**

Create `apps/api/internal/config/crypto.go`:

```go
package config

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"encoding/hex"
	"fmt"
	"io"
)

// Encrypt encrypts plaintext using AES-GCM with the hex-encoded key.
func Encrypt(plaintext string, hexKey string) (string, error) {
	key, err := hex.DecodeString(hexKey)
	if err != nil {
		return "", fmt.Errorf("decoding encryption key: %w", err)
	}
	block, err := aes.NewCipher(key)
	if err != nil {
		return "", fmt.Errorf("creating cipher: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return "", fmt.Errorf("creating GCM: %w", err)
	}
	nonce := make([]byte, gcm.NonceSize())
	if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
		return "", fmt.Errorf("generating nonce: %w", err)
	}
	ciphertext := gcm.Seal(nonce, nonce, []byte(plaintext), nil)
	return hex.EncodeToString(ciphertext), nil
}

// Decrypt decrypts hex-encoded ciphertext using AES-GCM with the hex-encoded key.
func Decrypt(ciphertextHex string, hexKey string) (string, error) {
	key, err := hex.DecodeString(hexKey)
	if err != nil {
		return "", fmt.Errorf("decoding encryption key: %w", err)
	}
	ciphertext, err := hex.DecodeString(ciphertextHex)
	if err != nil {
		return "", fmt.Errorf("decoding ciphertext: %w", err)
	}
	block, err := aes.NewCipher(key)
	if err != nil {
		return "", fmt.Errorf("creating cipher: %w", err)
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return "", fmt.Errorf("creating GCM: %w", err)
	}
	nonceSize := gcm.NonceSize()
	if len(ciphertext) < nonceSize {
		return "", fmt.Errorf("ciphertext too short")
	}
	nonce, ciphertext := ciphertext[:nonceSize], ciphertext[nonceSize:]
	plaintext, err := gcm.Open(nil, nonce, ciphertext, nil)
	if err != nil {
		return "", fmt.Errorf("decrypting: %w", err)
	}
	return string(plaintext), nil
}
```

```typescript
import { runPass1, type Message } from "./pass1";
import { runPass2, type StoryWithMessages } from "./pass2";

// Anthropic pricing per million tokens (as of 2025)
const PRICING: Record<string, { input: number; output: number }> = {
  "claude-haiku-4-5-20251001": { input: 0.80, output: 4.00 },
  "claude-sonnet-4-6": { input: 3.00, output: 15.00 },
};

function calculateCost(
  model: string,
  tokensIn: number,
  tokensOut: number
): number {
  const pricing = PRICING[model] ?? { input: 3.0, output: 15.0 };
  return (
    (tokensIn / 1_000_000) * pricing.input +
    (tokensOut / 1_000_000) * pricing.output
  );
}

export interface PipelineInput {
  communityType: string;
  serverName: string;
  messages: Message[];
}

export interface PipelineOutput {
  markdown: string;
  costUsd: number;
  pass1TokensIn: number;
  pass1TokensOut: number;
  pass2TokensIn: number;
  pass2TokensOut: number;
}

export async function runPipeline(
  input: PipelineInput
): Promise<PipelineOutput> {
  // Pass 1: Rank and identify stories
  const pass1 = await runPass1(input.messages, input.communityType);

  // Take top 8 stories by engagement_signal
  const topStories = pass1.output.stories
    .sort((a, b) => b.engagement_signal - a.engagement_signal)
    .slice(0, 8);

  // Re-attach source messages to each story
  const messageMap = new Map(input.messages.map((m) => [m.id, m]));
  const storiesWithMessages: StoryWithMessages[] = topStories.map((story) => ({
    ...story,
    messages: story.key_message_ids
      .map((id) => messageMap.get(id))
      .filter((m): m is Message => m !== undefined),
  }));

  // Pass 2: Draft newsletter
  const pass2 = await runPass2(storiesWithMessages, input.serverName);

  const model1 = process.env.AI_MODEL_PASS1 ?? "claude-haiku-4-5-20251001";
  const model2 = process.env.AI_MODEL_PASS2 ?? "claude-sonnet-4-6";

  const costUsd =
    calculateCost(model1, pass1.tokensIn, pass1.tokensOut) +
    calculateCost(model2, pass2.tokensIn, pass2.tokensOut);

  return {
    markdown: pass2.markdown,
    costUsd,
    pass1TokensIn: pass1.tokensIn,
    pass1TokensOut: pass1.tokensOut,
    pass2TokensIn: pass2.tokensIn,
    pass2TokensOut: pass2.tokensOut,
  };
}

export type { Message } from "./pass1";
```

```go
package llmclient

import (
	"bytes"
	"context"
	"encoding/json"

	"io"
	"net/http"
	"time"

// Client is a thin HTTP wrapper that calls the Next.js LLM endpoint.
type Client struct {
	baseURL    string
	apiKey     string
	httpClient *http.Client
}

// New creates a new LLM client.
func New(baseURL, apiKey string) *Client {
	return &Client{
		baseURL: baseURL,
		apiKey:  apiKey,
		httpClient: &http.Client{
			Timeout: 120 * time.Second,
		},
	}
}

// GenerateRequest is the payload sent to /api/internal/generate.
type GenerateRequest struct {
	CommunityType string    `json:"community_type"`
	ServerName    string    `json:"server_name"`
	Messages      []Message `json:"messages"`
}

// Message matches the TypeScript Message type.
type Message struct {
	ID            string `json:"id"`
	AuthorID      string `json:"authorId"`
	AuthorName    string `json:"authorName"`
	Content       string `json:"content"`
	Timestamp     string `json:"timestamp"`
	ReactionCount int    `json:"reactionCount"`
	ReplyCount    int    `json:"replyCount"`
	ReplyToID     string `json:"replyToId,omitempty"`
	ThreadID      string `json:"threadId,omitempty"`
	ChannelName   string `json:"channelName,omitempty"`
}

// GenerateResponse is the response from /api/internal/generate.
type GenerateResponse struct {
	Markdown      string  `json:"markdown"`
	CostUsd       float64 `json:"costUsd"`
	Pass1TokensIn int     `json:"pass1TokensIn"`
	Pass1TokensOut int    `json:"pass1TokensOut"`
	Pass2TokensIn int     `json:"pass2TokensIn"`
	Pass2TokensOut int    `json:"pass2TokensOut"`
}

// Generate calls the Next.js LLM pipeline endpoint.
func (c *Client) Generate(ctx context.Context, req GenerateRequest) (*GenerateResponse, error) {
	body, err := json.Marshal(req)
	if err != nil {
		return nil, fmt.Errorf("marshaling request: %w", err)
	}

	httpReq, err := http.NewRequestWithContext(ctx, http.MethodPost, c.baseURL+"/api/internal/generate", bytes.NewReader(body))
	if err != nil {
		return nil, fmt.Errorf("creating request: %w", err)
	}
	httpReq.Header.Set("Content-Type", "application/json")
	httpReq.Header.Set("Authorization", "Bearer "+c.apiKey)

	resp, err := c.httpClient.Do(httpReq)
	if err != nil {
		return nil, fmt.Errorf("calling generate endpoint: %w", err)
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		respBody, _ := io.ReadAll(resp.Body)
		return nil, fmt.Errorf("generate endpoint returned %d: %s", resp.StatusCode, string(respBody))
	}

	var result GenerateResponse
	if err := json.NewDecoder(resp.Body).Decode(&result); err != nil {
		return nil, fmt.Errorf("decoding response: %w", err)
	}
	return &result, nil
}

// PublishRequest is the payload sent to /api/internal/publish.
type PublishRequest struct {

	Subject       string `json:"subject"`
	Platform      string `json:"platform"`
```

```go
		"inserted", inserted,
	)
	return nil
```

```go
		Markdown:      markdown,
		Subject:       job.Args.Subject,
		Platform:      job.Args.Platform,
		APIKey:        apiKey,

	})

		errMsg := err.Error()

			ID:           job.Args.NewsletterID,
			Status:       "failed",
```

```go
	"log/slog"

	"github.com/sislelabs/distill/apps/api/internal/db"
```

```go
// SchedulerArgs are the args for the hourly scheduler check.
type SchedulerArgs struct{}

func (SchedulerArgs) Kind() string { return "scheduler" }

// SchedulerWorker checks which servers should have newsletters generated.
type SchedulerWorker struct {
	river.WorkerDefaults[SchedulerArgs]
	queries *db.Queries
	client  *river.Client[any]
}

func (w *SchedulerWorker) Work(ctx context.Context, job *river.Job[SchedulerArgs]) error {
	servers, err := w.queries.ListActiveServersForSchedule(ctx)
	if err != nil {
		return fmt.Errorf("listing active servers: %w", err)
	}

	now := time.Now()
	enqueued := 0

	for _, s := range servers {
		if !cronMatchesNow(s.ScheduleCron, now) {
			continue
		}

		_, err := w.client.Insert(ctx, GenerateNewsletterArgs{
			ServerID: s.ID,
		}, nil)
		if err != nil {
			slog.Warn("failed to enqueue generate job", "server_id", s.ID, "err", err)
			continue
		}
		enqueued++
	}

	slog.Info("scheduler run complete", "checked", len(servers), "enqueued", enqueued)
	return nil
}

// cronMatchesNow does a simple check: parses "minute hour * * dayOfWeek"
// and checks if current time matches. Only supports the format used by Distill.
func cronMatchesNow(cron string, now time.Time) bool {
	var minute, hour, dayOfWeek int
	_, err := fmt.Sscanf(cron, "%d %d * * %d", &minute, &hour, &dayOfWeek)
	if err != nil {
		slog.Warn("invalid cron expression", "cron", cron, "err", err)
		return false
	}

	// Check if we're in the matching hour (scheduler runs hourly)
	return now.Hour() == hour && int(now.Weekday()) == dayOfWeek
}
```

```go
// Fetcher fetches message history from Discord REST API.
type Fetcher struct {
	session *discordgo.Session
}

// NewFetcher creates a new Discord fetcher.
func NewFetcher(session *discordgo.Session) *Fetcher {
	return &Fetcher{session: session}
}

// FetchChannelHistory fetches messages from a channel for the given number of days.
func (f *Fetcher) FetchChannelHistory(guildID, channelID string, days int) ([]jobs.DiscordHistoryMessage, error) {
	cutoff := time.Now().AddDate(0, 0, -days)
	var allMessages []jobs.DiscordHistoryMessage
	beforeID := ""

	for {
		msgs, err := f.session.ChannelMessages(channelID, 100, beforeID, "", "")
		if err != nil {
			return nil, fmt.Errorf("fetching messages: %w", err)
		}

		if len(msgs) == 0 {
			break
		}

		for _, m := range msgs {
			ts, err := m.Timestamp.Parse()
			if err != nil {
				continue
			}

			if ts.Before(cutoff) {
				return allMessages, nil
			}

			raw, _ := json.Marshal(m)

			msg := jobs.DiscordHistoryMessage{
				ID:            m.ID,
				AuthorID:      m.Author.ID,
				AuthorName:    m.Author.Username,
				Content:       m.Content,
				Timestamp:     ts.Format(time.RFC3339),
				ReactionCount: countReactions(m.Reactions),
				RawPayload:    raw,
			}
			if m.MessageReference != nil {
				msg.ReplyToID = m.MessageReference.MessageID
			}
			if m.Thread != nil {
				msg.ThreadID = m.Thread.ID
			}

			allMessages = append(allMessages, msg)
		}

		beforeID = msgs[len(msgs)-1].ID
		slog.Info("fetched page", "channel", channelID, "count", len(msgs), "total", len(allMessages))

		// Respect rate limits
		time.Sleep(500 * time.Millisecond)
	}

	return allMessages, nil
}

func countReactions(reactions []*discordgo.MessageReactions) int {
	total := 0
	for _, r := range reactions {
		total += r.Count
	}
	return total
}
```

```go
package auth

import (
	"context"
	"net/http"
	"strings"

	"github.com/google/uuid"
)

type contextKey string

const UserIDKey contextKey = "user_id"

// Middleware validates the session token and sets user ID in context.
// For v1, this validates a simple bearer token (the NextAuth session token)
// passed by the Next.js frontend as a proxy.
func Middleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		auth := r.Header.Get("Authorization")
		if auth == "" {
			http.Error(w, "unauthorized", http.StatusUnauthorized)
			return
		}

		token := strings.TrimPrefix(auth, "Bearer ")
		// The token is the user ID passed by the Next.js BFF layer
		userID, err := uuid.Parse(token)
		if err != nil {
			http.Error(w, "invalid token", http.StatusUnauthorized)
			return
		}

		ctx := context.WithValue(r.Context(), UserIDKey, userID)
		next.ServeHTTP(w, r.WithContext(ctx))
	})
}

// GetUserID extracts the user ID from context.
func GetUserID(ctx context.Context) uuid.UUID {
	id, _ := ctx.Value(UserIDKey).(uuid.UUID)
	return id
}
```

```go
type Handler struct {
	queries       *db.Queries
	pool          *pgxpool.Pool
	riverClient   *river.Client[any]
	cfg           *config.Config
}

// NewRouter creates the chi router with all routes.
func NewRouter(pool *pgxpool.Pool, queries *db.Queries, riverClient *river.Client[any], cfg *config.Config) chi.Router {
	h := &Handler{
		queries:     queries,
		pool:        pool,
		riverClient: riverClient,

	}

	r.Use(middleware.Logger)

	r.Use(corsMiddleware)

	// Authenticated routes
	r.Group(func(r chi.Router) {
		r.Use(auth.Middleware)

		r.Get("/api/me", h.getMe)

		r.Get("/api/servers", h.listServers)
		r.Get("/api/servers/{id}", h.getServer)
		r.Patch("/api/servers/{id}", h.updateServer)
		r.Post("/api/servers/{id}/channels", h.addChannel)
		r.Delete("/api/servers/{id}/channels/{channelId}", h.removeChannel)
		r.Post("/api/servers/{id}/generate", h.triggerGenerate)

		r.Get("/api/newsletters", h.listNewsletters)
		r.Get("/api/newsletters/{id}", h.getNewsletter)
		r.Patch("/api/newsletters/{id}", h.updateNewsletter)
		r.Post("/api/newsletters/{id}/publish", h.publishNewsletter)

		r.Get("/api/integrations", h.listIntegrations)
		r.Post("/api/integrations/{platform}", h.connectIntegration)
		r.Delete("/api/integrations/{platform}", h.disconnectIntegration)

		r.Post("/api/billing/checkout", h.createCheckout)
		r.Post("/api/billing/portal", h.createPortal)
	})

	return r
}

func corsMiddleware(next http.Handler) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.Header().Set("Access-Control-Allow-Origin", "*")
		w.Header().Set("Access-Control-Allow-Methods", "GET, POST, PATCH, DELETE, OPTIONS")
		w.Header().Set("Access-Control-Allow-Headers", "Content-Type, Authorization")
		if r.Method == "OPTIONS" {
			w.WriteHeader(http.StatusOK)
			return
		}
		next.ServeHTTP(w, r)
	})
}
```

```go
package http

import (
	"encoding/json"
	"io"
	"log/slog"
	"net/http"

	"github.com/google/uuid"
	"github.com/stripe/stripe-go/v78"

	"github.com/sislelabs/distill/apps/api/internal/db"
)

func (h *Handler) handleStripeWebhook(w http.ResponseWriter, r *http.Request) {
	body, err := io.ReadAll(r.Body)
	if err != nil {
		http.Error(w, "failed to read body", http.StatusBadRequest)
		return
	}

	event, err := webhook.ConstructEvent(body, r.Header.Get("Stripe-Signature"), h.cfg.StripeWebhookSecret)
	if err != nil {
		http.Error(w, "invalid signature", http.StatusBadRequest)
		return
	}

	switch event.Type {
	case "customer.subscription.created", "customer.subscription.updated":
		var sub stripe.Subscription
		if err := json.Unmarshal(event.Data.Raw, &sub); err != nil {
			slog.Error("failed to unmarshal subscription", "err", err)
			break
		}
		h.handleSubscriptionChange(r, sub)

	case "customer.subscription.deleted":
		var sub stripe.Subscription
		if err := json.Unmarshal(event.Data.Raw, &sub); err != nil {
			slog.Error("failed to unmarshal subscription", "err", err)
			break
		}
		user, err := h.queries.GetUserByStripeCustomerID(r.Context(), &sub.Customer.ID)
		if err != nil {
			slog.Error("user not found for customer", "customer_id", sub.Customer.ID)
			break
		}
		h.queries.UpdateSubscriptionStatus(r.Context(), db.UpdateSubscriptionStatusParams{
			ID:                 user.ID,
			SubscriptionStatus: "canceled",
		})

	case "invoice.payment_failed":
		var inv stripe.Invoice
		if err := json.Unmarshal(event.Data.Raw, &inv); err != nil {
			slog.Error("failed to unmarshal invoice", "err", err)
			break
		}
		user, err := h.queries.GetUserByStripeCustomerID(r.Context(), &inv.Customer.ID)
		if err != nil {
			slog.Error("user not found for customer", "customer_id", inv.Customer.ID)
			break
		}
		h.queries.UpdateSubscriptionStatus(r.Context(), db.UpdateSubscriptionStatusParams{
			ID:                 user.ID,
			SubscriptionStatus: "past_due",
		})
	}

	w.WriteHeader(http.StatusOK)
}

func (h *Handler) handleSubscriptionChange(r *http.Request, sub stripe.Subscription) {
	customerID := sub.Customer.ID

	user, err := h.queries.GetUserByStripeCustomerID(r.Context(), &customerID)
	if err != nil {
		// Try to find by client_reference_id (set during checkout)
		if sub.Metadata != nil {
			if uid, ok := sub.Metadata["user_id"]; ok {
				userID, _ := uuid.Parse(uid)
				h.queries.UpdateStripeCustomerID(r.Context(), db.UpdateStripeCustomerIDParams{
					ID:               userID,
					StripeCustomerID: &customerID,
				})
				user, err = h.queries.GetUserByID(r.Context(), userID)
			}
		}
		if err != nil {
			slog.Error("user not found for subscription", "customer_id", customerID)
			return
		}
	}

	status := "active"
	switch sub.Status {
	case stripe.SubscriptionStatusActive:
		status = "active"
	case stripe.SubscriptionStatusPastDue:
		status = "past_due"
	case stripe.SubscriptionStatusCanceled:
		status = "canceled"
	case stripe.SubscriptionStatusTrialing:
		status = "trialing"
	default:
		status = string(sub.Status)
	}

	h.queries.UpdateSubscriptionStatus(r.Context(), db.UpdateSubscriptionStatusParams{
		ID:                 user.ID,
		SubscriptionStatus: status,
	})
}
```

```go
package main

import (
	"context"
	"log/slog"
	"net/http"
	"os"
	"os/signal"
	"syscall"
	"time"

	"github.com/sislelabs/distill/apps/api/internal/db"
	apphttp "github.com/sislelabs/distill/apps/api/internal/http"
)

func main() {
	cfg, err := config.Load()
	if err != nil {
		slog.Error("failed to load config", "err", err)
		os.Exit(1)
	}

	ctx := context.Background()
	pool, err := pgxpool.New(ctx, cfg.DatabaseURL)
	if err != nil {
		slog.Error("failed to connect to database", "err", err)
		os.Exit(1)
	}
	defer pool.Close()

	queries := db.New(pool)

	router := apphttp.NewRouter(pool, queries, nil, cfg)

	srv := &http.Server{
		Addr:    ":8080",
		Handler: router,
	}

	go func() {
		slog.Info("API server starting", "addr", ":8080")
		if err := srv.ListenAndServe(); err != nil && err != http.ErrServerClosed {
			slog.Error("server error", "err", err)
			os.Exit(1)
		}
	}()

	stop := make(chan os.Signal, 1)
	signal.Notify(stop, syscall.SIGINT, syscall.SIGTERM)
	<-stop

	slog.Info("shutting down API server")
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()
	srv.Shutdown(ctx)
}
```

```go
		slog.Error("failed to load config", "err", err)
		os.Exit(1)
	}

	ctx := context.Background()
	pool, err := pgxpool.New(ctx, cfg.DatabaseURL)
	if err != nil {
		slog.Error("failed to connect to database", "err", err)
		os.Exit(1)
	}
	defer pool.Close()

	queries := db.New(pool)
	llm := llmclient.New(cfg.WebInternalBaseURL, cfg.InternalAPIKey)

	workers := river.NewWorkers()

	// Register all job workers
	river.AddWorker(workers, &jobs.GenerateNewsletterWorker{})
	river.AddWorker(workers, &jobs.BackfillChannelWorker{})
	river.AddWorker(workers, &jobs.PublishNewsletterWorker{})
	river.AddWorker(workers, &jobs.TrialReminderWorker{})
	river.AddWorker(workers, &jobs.SchedulerWorker{})

	riverClient, err := river.NewClient(riverpgxv5.New(pool), &river.Config{
		Queues: map[string]river.QueueConfig{
			river.QueueDefault: {MaxWorkers: 5},
		},
		Workers: workers,
	})
	if err != nil {
		slog.Error("failed to create river client", "err", err)
		os.Exit(1)
	}

	// Start periodic jobs
	// Note: River's periodic job setup depends on exact API version;
	// the scheduler and trial_reminder run as periodic jobs.
	// We'll configure them via River's PeriodicJobs config.

	if err := riverClient.Start(ctx); err != nil {
		slog.Error("failed to start river client", "err", err)
		os.Exit(1)
	}

	_ = queries
	_ = llm

	slog.Info("worker started")

	stop := make(chan os.Signal, 1)
	signal.Notify(stop, syscall.SIGINT, syscall.SIGTERM)
	<-stop

	slog.Info("shutting down worker")
	riverClient.Stop(ctx)
}
```

```typescript
import type { Publisher, PublishRequest, PublishResult } from "./types";

export class GhostPublisher implements Publisher {
  async publish(req: PublishRequest): Promise<PublishResult> {
    // Ghost Admin API requires JWT signed with the admin API key
    const [id, secret] = req.apiKey.split(":");

    // Create JWT for Ghost Admin API
    const header = Buffer.from(JSON.stringify({ alg: "HS256", kid: id, typ: "JWT" })).toString("base64url");
    const now = Math.floor(Date.now() / 1000);
    const payload = Buffer.from(JSON.stringify({ iat: now, exp: now + 300, aud: "/admin/" })).toString("base64url");

    const crypto = await import("crypto");
    const signature = crypto
      .createHmac("sha256", Buffer.from(secret, "hex"))
      .update(`${header}.${payload}`)
      .digest("base64url");
    const token = `${header}.${payload}.${signature}`;

    const apiUrl = req.publicationId; // Ghost uses the blog URL as publication ID
    const response = await fetch(`${apiUrl}/ghost/api/admin/posts/`, {
      method: "POST",
      headers: {
        Authorization: `Ghost ${token}`,
        "Content-Type": "application/json",
      },
      body: JSON.stringify({
        posts: [
          {
            title: req.subject,
            mobiledoc: JSON.stringify({
              version: "0.3.1",
              atoms: [],
              cards: [["markdown", { markdown: req.markdown }]],
              markups: [],
              sections: [[10, 0]],
            }),
            status: "draft",
          },
        ],
      }),
    });

    if (!response.ok) {
      const text = await response.text();
      throw new Error(`Ghost API error ${response.status}: ${text}`);
    }

    const data = await response.json();
```

```typescript
export const { handlers, signIn, signOut, auth } = NextAuth({
  providers: [
    Discord({
      clientId: process.env.DISCORD_CLIENT_ID!,

      // Upsert user into our database
      const discordId = account.providerAccountId;
      const username = (profile as any)?.username ?? user.name ?? "unknown";
      const email = user.email ?? "";
      const avatar = user.image ?? null;

      const result = await pool.query(
        `INSERT INTO users (discord_id, discord_username, email, avatar_url)
         VALUES ($1, $2, $3, $4)
         ON CONFLICT (discord_id) DO UPDATE SET
           discord_username = EXCLUDED.discord_username,
           email = EXCLUDED.email,
           avatar_url = EXCLUDED.avatar_url,
           updated_at = NOW()
         RETURNING id, subscription_status, trial_ends_at`,
        [discordId, username, email, avatar]
      );

      return true;
    },
    async jwt({ token, account, profile }) {
      if (account) {
        token.discordId = account.providerAccountId;

        // Fetch our internal user ID
        const result = await pool.query(
          "SELECT id, subscription_status, trial_ends_at FROM users WHERE discord_id = $1",
          [account.providerAccountId]
        );
        if (result.rows.length > 0) {
          token.userId = result.rows[0].id;
          token.subscriptionStatus = result.rows[0].subscription_status;
          token.trialEndsAt = result.rows[0].trial_ends_at;
        }

        // Store access token for Discord API calls
        token.accessToken = account.access_token;
      }
      return token;
    },
    async session({ session, token }) {
      (session as any).userId = token.userId;
      (session as any).discordId = token.discordId;
      (session as any).subscriptionStatus = token.subscriptionStatus;
      (session as any).trialEndsAt = token.trialEndsAt;
      (session as any).accessToken = token.accessToken;
      return session;
    },
  },
```
